## Prix du jour
import "date"

from(bucket:"price")
|> range(start: today(), stop: date.add(d: 24h, to: today()))
|> filter(fn:(r) => r._measurement == "cout_electricite")
|> filter(fn:(r) => r.location == "France")
|> filter(fn:(r) => r._field == "euros")
